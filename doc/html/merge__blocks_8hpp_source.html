86 /// @brief make the indirect merge of the two range_pos inside RP
87 /// @param [in] pf : pointer to the atomic counter of the father
88 /// @param [in] RP : rnage_pos which contains the two ranges to merge
89 /// @param [in] Rbuf : auxiliary memory for to do the merge
90 //-------------------------------------------------------------------------
91 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
92 merge_blocks < block_size, group_size, iter_t, compare>::
93 merge_blocks ( backbone_t &bkb, size_t pos_index1, size_t pos_index2,
94  size_t pos_index3 ):bk(bkb),counter(0)
95 { //----------------------------- begin --------------------------------
96  size_t nblock1 = pos_index2 - pos_index1 ;
97  size_t nblock2 = pos_index3 - pos_index2 ;
98  if ( nblock1 == 0 or nblock2 == 0 ) return ;
99 
100  //-----------------------------------------------------------------------
101  // Merging of the two intervals
102  //-----------------------------------------------------------------------
103  std::vector<block_pos> vpos1, vpos2 ;
104  vpos1.reserve ( nblock1 +1 );
105  vpos2.reserve ( nblock2 +1 );
106 
107  for ( size_t i = pos_index1 ; i < pos_index2 ; ++i)
108  { vpos1.emplace_back( bk.index[i].pos(), true) ;
109  };
110 
111  for ( size_t i = pos_index2; i < pos_index3 ; ++i)
112  { vpos2.emplace_back( bk.index[i].pos(), false) ;
113  };
114  //-------------------------------------------------------------------
115  // tail process
116  //-------------------------------------------------------------------
117  if ( vpos2.back().pos() == (bk.nblock -1) and
118  bk.range_tail.first != bk.range_tail.last)
119  { tail_process ( vpos1, vpos2);
120  nblock1 = vpos1.size();
121  nblock2 = vpos2.size() ;
122  };
123 
124  compare_block_pos_t cmp_blk ( bk.global_range.first, bk.cmp) ;
125 
126  util::full_merge (vpos1.begin() , vpos1.end(),
127  vpos2.begin() , vpos2.end(),
128  bk.index.begin() + pos_index1, cmp_blk);
129 
130  //--------------------------------------------------------------------
131  // Extracting the ranges for to merge the elements
132  //--------------------------------------------------------------------
133  std::vector <range_pos> vrange_pos ;
134  cut_range (range_pos(pos_index1,pos_index1+nblock1+nblock2),vrange_pos);
135 
136  //-------------------------------------------------------------------
137  // parallel process
138  //-------------------------------------------------------------------
139  util::atomic_write ( counter , vrange_pos.size());
140  for ( size_t i =0 ; i < vrange_pos.size() ; ++i)
141  { function_t f1=[=, &vrange_pos ]( )
142  { merge_range_pos ( vrange_pos[i]);
143  util::atomic_sub ( this->counter , 1);
144  };
145  bk.works.emplace_back ( f1 );
146  };
147  bk.exec ( counter) ; // wait until finish all the ranges
148 };

153 /// @brief make the indirect merge of the two range_pos inside RP

158 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
159 void merge_blocks < block_size, group_size, iter_t, compare>::
160 tail_process ( std::vector<block_pos> & vblkpos1,
161  std::vector<block_pos> & vblkpos2 )
162 { //-------------------------- begin ---------------------------------
163  if ( vblkpos1.size() == 0 or vblkpos2.size() == 0 ) return ;
164 
165  vblkpos2.pop_back();
166 
167  size_t posback1 = vblkpos1.back().pos() ;
168  range_it range_back1 = bk.get_range (posback1 );
169 
170  if ( util::is_mergeable (range_back1,bk.range_tail, bk.cmp))
171  { util::in_place_merge_uncontiguous ( range_back1, bk.range_tail,
172  bk.get_range_buf(), bk.cmp);
173  if ( vblkpos1.size() >1 )
174  { //---------------------------------------------------------------
175  size_t pos_aux = vblkpos1[ vblkpos1.size() -2].pos() ;
176  range_it range_aux = bk.get_range ( pos_aux) ;
177 
178  if ( util::is_mergeable( range_aux,range_back1, bk.cmp))
179  { vblkpos2.emplace_back( posback1, false ) ;
180  vblkpos1.pop_back() ;
181  };
182  };
183  };

189 /// @brief divide the range_pos rng in several ranges. The ranges obtained
190 /// are inserted in the vector of range_pos vrpos_out
191 /// @param [in] rng : range to divide
192 /// @param [in] vrpos_out : vector where insert the ranges obtained in the
193 /// division of rng
194 //-------------------------------------------------------------------------
195 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
196 void merge_blocks < block_size, group_size, iter_t, compare>::
197 cut_range (range_pos rng , std::vector <range_pos> & vrpos_out)
198 { //---------------------------- begin -----------------------------
199  if ( rng.size() < group_size )
200  { vrpos_out.push_back ( rng);
201  return ;
202  };
203  size_t npart = (rng.size() + group_size -1 )/ group_size ;
204  size_t size_part = rng.size() / npart ;
205 
206  size_t pos_ini = rng.first ;
207  size_t pos_last = rng.last ;
208  while ( pos_ini < pos_last)
209  { size_t pos = pos_ini + size_part ;
210  while ( pos < pos_last and
211  bk.index[pos-1].side() == bk.index[pos].side())
212  { ++pos ;
213  };
214  if ( pos < pos_last)
215  { in_place_merge_uncontiguous
216  ( bk.get_range ( bk.index[pos-1].pos()),
217  bk.get_range ( bk.index[pos].pos() ),
218  bk.get_range_buf(), bk.cmp);
219  }
220  else pos = pos_last ;
221  if ( (pos - pos_ini) > 1 )
222  vrpos_out.emplace_back( pos_ini, pos);
223  pos_ini = pos ;
224  };
225 };

235 merge_range_pos ( range_pos rng )
236 { //------------------------- begin ------------------------------------
237  if ( rng.size() < 2 ) return ;
238  range_buf rbuf = bk.get_range_buf () ;
239 
240  range_it rng_prev = bk.get_range ( bk.index[ rng.first ].pos());
241  init_move ( rbuf, rng_prev);
242  range_it rng_posx (rng_prev);
243 
244  for (size_t posx = rng.first +1 ; posx != rng.last ; ++posx )
245  { rng_posx = bk.get_range ( bk.index[posx].pos());
246  util::merge_flow ( rng_prev, rbuf, rng_posx, bk.cmp);
247  rng_prev = rng_posx;
248 
249  }; // end while
250  init_move (rng_posx, rbuf);
251 }; // end merge_range_pos ( range_pos rng )
