84 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
85 move_blocks <block_size, group_size, iter_t, compare>::
86 move_blocks ( backbone_t &bkb ):bk(bkb),counter(0)
87 { //------------------------- begin -------------------------------------
88  std::vector < std::vector <size_t> > vsequence ;
89 
90  std::vector<size_t> sequence ;
91  size_t pos_index_ini =0 , pos_index_src =0, pos_index_dest =0 ;
92  while ( pos_index_ini < bk.index.size() )
93  { while (pos_index_ini < bk.index.size() and
94  bk.index[pos_index_ini].pos() == pos_index_ini)
95  { ++pos_index_ini;
96  };
97  if ( pos_index_ini == bk.index.size() ) break ;
98 
99  sequence.clear() ;
100  pos_index_src = pos_index_dest = pos_index_ini ;
101  sequence.push_back (pos_index_ini);
102 
103  while ( bk.index[ pos_index_dest].pos() != pos_index_ini )
104  { pos_index_src = bk.index[ pos_index_dest].pos();
105  sequence.push_back ( pos_index_src) ;
106 
107  bk.index[pos_index_dest].pos( pos_index_dest);
108  pos_index_dest = pos_index_src ;
109  };
110  bk.index[ pos_index_dest].pos( pos_index_dest);
111  vsequence.push_back( sequence);
112  };
113  util::atomic_write ( counter , vsequence.size() );
114 
115  //-------------------------------------------------------------------
116  // creation of the works
117  //-------------------------------------------------------------------
118  atomic_t son_counter ( vsequence.size() ) ;
119  for ( size_t i =0 ; i < vsequence.size() ; ++i )
120  { if ( vsequence[i].size() < group_size )
121  { function_t f1 = [ = , &vsequence, &son_counter]( )
122  { move_sequence(vsequence[i] );
123  util::atomic_sub ( son_counter, 1 ) ;
124  };
125  bk.works.emplace_back ( f1 ) ;
126  }
127  else

129  { move_long_sequence(vsequence[i]) ;

131  };
132  bk.works.emplace_back ( f1 ) ;

172 void move_blocks <block_size, group_size, iter_t, compare>::
173 move_long_sequence ( std::vector<size_t> & seq )
174 { //--------------------------------- begin ----------------------------
175  if ( seq.size() < group_size )
176  return move_sequence ( seq) ;
177 
178  size_t npart = ( seq.size() + group_size -1) / group_size ;
179  size_t size_part = seq.size() / npart ;
180 
181  // vectors for the sequences and the final sequence
182  std::vector< std::vector < size_t> > vseq ;
183  vseq.reserve ( npart ) ;
184 
185  std::vector < size_t> index_seq ;
186  index_seq.reserve ( npart ) ;
187 
188  auto it_pos = seq.begin() ;
189  for ( size_t i =0 ; i < (npart -1) ; ++i, it_pos += size_part)
190  { vseq.emplace_back ( it_pos, it_pos + size_part) ;
191  index_seq.emplace_back ( * ( it_pos+ size_part -1) ) ;
192  };
193  vseq.emplace_back ( it_pos , seq.end() ) ;
194  index_seq.emplace_back ( seq.back() ) ;
195 
196  // creation of the functions

198  for ( size_t i =0 ; i < vseq.size() ; ++i )
199  { function_t f1=[=, &vseq, &son_counter]()
200  { move_sequence( vseq[i] ) ;
201  util::atomic_sub ( son_counter, 1 );
202  };
203  bk.works.emplace_back (f1 ) ;
204  };
205  bk.exec ( son_counter) ;
206  move_long_sequence ( index_seq ) ;
