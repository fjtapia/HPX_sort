137 // function : block_indirect_sort
138 /// @brief begin with the execution of the functions stored in works
139 /// @param [in] counter : atomic counter. When 0 exits the function
140 /// @return range
141 //-------------------------------------------------------------------------
142 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
143 block_indirect_sort<block_size, group_size, iter_t, compare>::
144 block_indirect_sort ( iter_t first , iter_t last , compare cmp, uint32_t nthr )
145  :bk (first, last, cmp), counter(0), ptr(nullptr),
146  construct (false ),nthread ( nthr)
147 { //-------------------------- begin -------------------------------------
148  assert ( (last-first) >= 0 ) ;
149  size_t nelem = size_t ( last - first ) ;
150  if ( nelem == 0 ) return ;
151 
152  //------------------- check if sort --------------------------------------
153  bool sorted = true ;
154  for ( iter_t it1 = first, it2 = first+1 ;
155  it2 != last and (sorted = not bk.cmp(*it2,*it1));it1 = it2++);
156  if (sorted) return ;
157 
158  //---------------- check if only single thread -----------------------
159  size_t nthreadmax = nelem / (block_size * group_size) + 1;
160  if (nthread > nthreadmax) nthread = (uint32_t) nthreadmax;
161 
162  uint32_t nbits_size = (util::nbits64(sizeof (value_t)))>>1;
163  if ( nbits_size > 5 ) nbits_size = 5 ;
164  size_t max_per_thread = 1<< (18 - nbits_size);
165 
166 
167  if ( nelem < ( max_per_thread) or nthread < 2)
168  { intro_sort (first, last, bk.cmp);
169  return ;
170  };
171 
172  //----------------- creation of the temporary buffer --------------------
173  ptr =std::get_temporary_buffer<value_t>( block_size*nthread).first ;
174  if ( ptr == nullptr) throw std::bad_alloc() ;
175 
176  rglobal_buf = range_buf ( ptr , ptr + (block_size * nthread) );
177  util::init ( rglobal_buf, *first);
178  construct = true ;
179 
180  std::vector< value_t*> vbuf ( nthread);
181  for ( uint32_t i =0 ; i < nthread ; ++i)
182  vbuf[i] = ptr + (i * block_size);
183 
184  //------------------------------------------------------------------------
185  util::atomic_write ( counter , 1 ) ;
186  function_t f1 = [&]()
187  { start_function( );
188  util::atomic_sub( counter , 1 ) ;
189  };
190  bk.works.emplace_back ( f1 ) ;
191 
192  //------------------------------------------------------------------------
193  // PROCESS
194  //------------------------------------------------------------------------
195  std::vector <hpx::future <void> > vfuture (nthread);
196 
197  for ( uint32_t i =0 ; i < nthread ; ++i)
198  { auto f1 = [=, &vbuf ] () { bk.exec (vbuf[i], this->counter);};
199  vfuture[i] = hpx::async ( f1 );
200  };
201  for ( uint32_t i =0 ; i < nthread ; ++i) vfuture[i].get() ;
202 };

207 // function : split_rage
208 /// @brief this function splits a range, and depending of the size, sort
209 /// directly or call to a recursive split_range
210 /// @param [in] pf : pointer to the atomic counter of the father
211 /// @param [in] RP : range of positions to split
212 /// @param [in] Level : depth of the split
213 //-----------------------------------------------------------------------------
214 template <uint32_t block_size, uint32_t group_size, class iter_t, class compare>
215 void block_indirect_sort<block_size, group_size, iter_t, compare>::
216 split_range ( size_t pos_index1 ,size_t pos_index2, uint32_t level_thread )
217 { //----------------------------- begin -----------------------------------
218  size_t nblock = pos_index2 - pos_index1;
219 
220  //-------------------------------------------------------------------------
221  // In the blocks not sorted, the physical position is the logical position
222  //-------------------------------------------------------------------------
223  iter_t first = bk.get_block ( pos_index1).first ;
224  iter_t last = bk.get_range ( pos_index2 -1).last ;
225 
226  if ( nblock < group_size)
227  { intro_sort ( first, last , bk.cmp ) ;
228  return ;
229  };
230 
231  size_t pos_index_mid = pos_index1 + (nblock >>1 ) ;
232  atomic_t son_counter (1) ;
233 
234  //-------------------------------------------------------------------------
235  // Insert in the stack the work for the second part, and the actual thread,
236  // execute the first part
237  //-------------------------------------------------------------------------
238  if ( level_thread != 0 )
239  { auto f1 = [=, &son_counter]( )
240  { split_range ( pos_index_mid,pos_index2, level_thread-1) ;
241  util::atomic_sub ( son_counter , 1 ) ;
242  };
243  bk.works.emplace_back ( f1 ) ;
244  split_range ( pos_index1 , pos_index_mid,level_thread -1) ;
245  }
246  else
247  { iter_t mid = first + ( (nblock >>1 ) * block_size ) ;
248  auto f1 = [=, &son_counter]()
249  { parallel_sort_t ( bk, mid , last ) ;
250  util::atomic_sub ( son_counter , 1 ) ;
251  };
252  bk.works.emplace_back ( f1 ) ;
253  parallel_sort_t ( bk , first ,mid ) ;
254  };
255  bk.exec ( son_counter ) ;
256  merge_blocks_t ( bk, pos_index1 , pos_index_mid , pos_index2 ) ;
257 };

277  move_blocks_t k( bk ) ;
