72 /// @brief constructor of the class
73 /// @param [in] bkbn : backbone struct with all the information to sort

77 template <uint32_t block_size, class iter_t, class compare>
78 parallel_sort <block_size, iter_t, compare>::
79 parallel_sort (backbone_t &bkbn, iter_t first, iter_t last):bk(bkbn), counter(0)
80 { //-------------------------- begin ------------------------------------
81  assert ( (last-first) >= 0 ) ;
82  size_t nelem = size_t ( last - first ) ;
83 
84  //------------------- check if sort --------------------------------------
85  bool sorted = true ;
86  for ( iter_t it1 = first, it2 = first+1 ;
87  it2 != last and (sorted = not bk.cmp(*it2,*it1));it1 = it2++);
88  if (sorted) return ;
89 
90  //-------------------max_per_thread ---------------------------
91  uint32_t nbits_size = (nbits64(sizeof (value_t)))>>1;
92  if ( nbits_size > 5 ) nbits_size = 5 ;
93  max_per_thread = 1<< (18 - nbits_size);
94  uint32_t level = ((nbits64(nelem / max_per_thread)) * 2);
95 
96  //---------------- check if only single thread -----------------------
97  if ( nelem < ( max_per_thread))
98  { intro_sort (first, last, bk.cmp);
99  return ;
100  };
101 
102  divide_sort ( first, last,level ) ;
103  // wait until all the parts are finished
104  bk.exec(counter);
105 };

109 /// @brief this function divide the data in two part, for to be sorted in
110 /// a parallel mode
111 /// @param [in] first : iterator to the first element to sort
112 /// @param [in] last : iterator to the next element after the last
113 //------------------------------------------------------------------------
114 template <uint32_t block_size, class iter_t, class compare>
115 void parallel_sort <block_size, iter_t, compare>::
116 divide_sort(iter_t first, iter_t last , uint32_t level )
117 { //------------------- check if sort -----------------------------------
118  bool sorted = true ;
119  for ( iter_t it1 = first, it2 = first+1 ;
120  it2 != last and (sorted = not bk.cmp(*it2,*it1));it1 = it2++);
121  if (sorted) return ;
122 
123  //---------------- check if finish the subdivision -------------------
124  size_t nelem = last - first ;
125  if ( level == 0 or nelem < ( max_per_thread))
126  { return intro_sort (first, last, bk.cmp);
127  };
128 
129  //-------------------- pivoting ----------------------------------
130  pivot9 ( first, last ,bk.cmp);
131  const value_t & val = const_cast < value_t &>(* first);
132  iter_t c_first = first+1 , c_last = last-1;
133 
134  while ( bk.cmp(*c_first, val) ) ++c_first ;
135  while ( bk.cmp ( val,*c_last ) ) --c_last ;
136  while (not( c_first > c_last ))
137  { std::swap ( *(c_first++), *(c_last--));
138  while ( bk.cmp (*c_first, val) ) ++c_first;
139  while ( bk.cmp ( val, *c_last) ) --c_last ;
140  }; // End while
141  std::swap ( *first , * c_last);
142 
143  // insert the work of the second half in the stack of works
144  util::atomic_add ( counter , 1 );
145  function_t f1 = [=] ( ){ divide_sort ( c_first, last,level -1) ;
146  util::atomic_sub( this->counter,1);
147  };
148  bk.works.emplace_back( f1) ;
149 
150  // The first half is done by the same thread
151  divide_sort (first, c_last, level -1);
152 };
